// --- OPTIMIERTE TOUCH & PENCIL LOGIK --- const getPos = (e) => { const offset = document.querySelector('.nav-bar').offsetHeight + document.querySelector('.tool-bar').offsetHeight; const touch = e.touches[0]; return { x: touch.clientX, y: touch.clientY - offset }; }; canvas.addEventListener('touchstart', (e) => { // PALM REJECTION: // 1. Wenn mehr als ein Kontaktpunkt da ist (Handballen + Stift), ignorieren wir den Rest. // 2. Wir priorisieren den 'stylus' (Apple Pencil). if (e.touches.length > 1 && e.touches[0].touchType !== 'stylus') { drawing = false; return; } drawing = true; ctx.beginPath(); const p = getPos(e); ctx.moveTo(p.x, p.y); // Verhindert das Scrollen der Seite während des Schreibens e.preventDefault(); }, {passive: false}); canvas.addEventListener('touchmove', (e) => { if (!drawing) return; // Wenn während des Schreibens ein zweiter Finger/Handballen dazu kommt: if (e.touches.length > 1 && e.touches[0].touchType !== 'stylus') return; const p = getPos(e); // Strich-Einstellungen für schöneres Schriftbild ctx.lineWidth = (e.touches[0].touchType === 'stylus') ? 2.0 : 2.5; ctx.lineCap = 'round'; ctx.lineJoin = 'round'; ctx.strokeStyle = '#222'; ctx.lineTo(p.x, p.y); ctx.stroke(); e.preventDefault(); }, {passive: false}); canvas.addEventListener('touchend', (e) => { drawing = false; saveCurrentPage(); });
